    <script>
        const mpesaModal = new bootstrap.Modal(document.getElementById('mpesaModal'));
        const API_BASE = window.location.origin;

        function openMpesaModal(service, amount) {
            document.getElementById('serviceName').value = service;
            document.getElementById('serviceAmount').value = 'KES ' + amount.toLocaleString();
            mpesaModal.show();
        }

        document.getElementById('mpesaForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            const name = document.getElementById('customerName').value;
            const phone = document.getElementById('customerPhone').value;
            const item = document.getElementById('serviceName').value;
            const amount = parseInt(document.getElementById('serviceAmount').value.replace(/[^0-9]/g, ''));

            try {
                const response = await fetch(API_BASE + '/api/create-order', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        name, email: 'info@business.co.ke', phone,
                        domain: window.location.hostname, item, price: amount, type: 'service'
                    })
                });
                const result = await response.json();
                if (result.success) {
                    alert('Payment request sent! Check your phone and enter M-Pesa PIN.');
                    mpesaModal.hide();
                    this.reset();
                } else {
                    alert('Error: ' + result.message);
                }
            } catch (error) {
                alert('Network error. Please try again.');
            }
        });
    </script>
