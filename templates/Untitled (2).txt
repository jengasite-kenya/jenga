<script>
let isLoggedIn=localStorage.getItem('adminLoggedIn')==='true';
if(isLoggedIn)showDashboard();
document.getElementById('loginForm').addEventListener('submit',function(e){e.preventDefault();if(document.getElementById('username').value==='admin'&&document.getElementById('password').value==='~admin_password~'){localStorage.setItem('adminLoggedIn','true');showDashboard();}else{alert('Invalid credentials');}});
function showDashboard(){document.getElementById('loginScreen').classList.add('hidden');document.getElementById('dashboard').classList.remove('hidden');}
function logout(){localStorage.removeItem('adminLoggedIn');location.reload();}
function showSection(section){['dashboard','products','orders','settings'].forEach(s=>document.getElementById(s+'Section').classList.add('hidden'));document.getElementById(section+'Section').classList.remove('hidden');const titles={dashboard:'Dashboard',products:'Products',orders:'Orders',settings:'Settings'};document.getElementById('pageTitle').textContent=titles[section];}
function openProductModal(){document.getElementById('productModal').classList.remove('hidden');}
function saveSettings(){const btn=event.target;const spinner=document.getElementById('saveSpinner');const pwd=document.getElementById('newPassword').value;const confirm=document.getElementById('confirmPassword').value;if(pwd&&pwd!==confirm){alert('Passwords do not match');return;}spinner.style.display='block';btn.disabled=true;setTimeout(()=>{spinner.style.display='none';btn.disabled=false;alert('Settings saved!');if(pwd){document.getElementById('newPassword').value='';document.getElementById('confirmPassword').value='';}},800);}
</script>
